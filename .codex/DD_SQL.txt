SELECT
    IA.DeptCode_Factory,
    IA.DEPTCODE,
    IA.CASEID,
    IA.ITEMNO,
    CS.MEETINGTYPE,
    CS.CaseName,
    CS.IsAllowReply,
    '項次' + CAST(CM.SEQNO AS VARCHAR(3)) + '.' SEQNO,
    CASE
        WHEN CM.ITEMTYPE = '0' THEN '<STRONG>持續列管</STRONG><BR>'
        ELSE ''
    END + CI.CONTENTS CONTENTS,
    IA.DEPTCONTENTS,
    CS.MEETINGTYPE + ',' + CAST(CM.CASEID AS VARCHAR(6)) + ',' + CAST(CM.ITEMNO AS VARCHAR(6)) ARG,
    IA.PROCMAN,
    SUBSTRING(IA.DTE_FINISH, 1, 3) + '.' + SUBSTRING(IA.DTE_FINISH, 4, 2) + '.' + SUBSTRING(IA.DTE_FINISH, 6, 2) + '.' + SUBSTRING(IA.DTE_FINISH, 8, 4) DTE_FINISH,  SUBSTRING(CS.DTE_DEADLINE, 1, 3) + '.' + SUBSTRING(CS.DTE_DEADLINE, 4, 2) + '.' + SUBSTRING(CS.DTE_DEADLINE, 6, 2) + '.' + SUBSTRING(CS.DTE_DEADLINE, 8, 4) DTE_DEADLINE,
    (
        SELECT
            SUBSTRING(DTE_DEADLINE, 1, 3) + '.' + SUBSTRING(DTE_DEADLINE, 4, 2) + '.' + SUBSTRING(DTE_DEADLINE, 6, 2) + '.' + SUBSTRING(DTE_DEADLINE, 8, 4)
        FROM
            FactoryHandleMan
        WHERE
            FACTORYID = '  Session(FactoryID)  '
            AND MEETINGTYPE = CS.MEETINGTYPE
    ) FACTORYDTE_DEADLINE
FROM
   CASES CS
    INNER JOIN CaseItem_Matches CM ON CS.CaseID = CM.CaseID
    INNER JOIN CASE_ITEMS CI ON CM.ITEMNO = CI.ITEMNO
    INNER JOIN ITEMS_ASSIGN IA ON CM.CASEID = IA.CASEID
    AND CM.ITEMNO = IA.ITEMNO
WHERE
    PROCMAN = @PROCMAN
    AND 1 = 1
    AND IA.ISFINISHED = '0'
ORDER BY
    IA.DEPTCODE, CS.MEETINGTYPE, CM.CASEID, CM.ITEMTYPE, CM.SEQNO

SELECT top 300
        TM.TM_GRSNO,
        CONVERT(VARCHAR(10), TM.TM_TDATE, 23) AS TM_TDATE,
        TM.TM_PSID,
        TM.TM_NAME,
--        EM.EM_SNO,
        TM.TM_RSTP,TM_RCDATE AS "核畢日期",
        TM_RCTYPE AS "核畢否",--(1.核畢 0.否 2.退件 3.加會 4.受會會辦重呈)
        TM_RPDT AS "核判日期",    
        OL.OR_STATUS AS "呈核狀態",  -- 處理狀態(0:原始檔;1:待處理;2:處理完成;3:退件;4:已核判)
        TM.TM_SUBJ AS "來文主旨/創案案由",
        TD.TD_FORMAT,        --文件種類
        DF.DF_NAME AS DF_NAME,
        DF.DF_DATA AS DF_DATA,
        DATALENGTH(DF.DF_DATA) AS DF_DATA_LEN,
--        EF.EF_ID AS EF_ID,
        EF.EF_NAME AS EF_NAME,
        EF.EF_DATA AS EF_DATA,
        DATALENGTH(EF.EF_DATA) AS EF_DATA_LEN,
        EF.EF_PAGE AS EF_PAGE,
        --CASE WHEN TD.TD_FORMAT in ('來文(檔案)') THEN  COALESCE(EF.EF_NAME,TD.TD_SUBJ)
        --     ELSE  COALESCE(TD.TD_SUBJ,EF.EF_NAME) END AS DOC_NAME,
        TD.TD_SUBJ AS TD_SUBJ
    FROM mnda.dbo.DCS3_TRST_MST TM                                       -- 呈核主檔
    LEFT JOIN mnda.dbo.DCS3_TRST_DAT TD ON TM.TM_SNO = TD.TD_SNO            -- 呈核文稿附件關連識別檔 
    LEFT JOIN mnda.dbo.DCS0_DOC_FILE DF ON DF.DF_PATH = TD.TD_PATH            -- 文稿實體檔
    LEFT JOIN mnda.dbo.DCS0_OLS_REC OL ON OR_GRSNO=TM.TM_GRSNO            -- 來文簽核檔
    LEFT JOIN mnda.dbo.DCS1_EMAL_TMP EM ON TM.TM_GRSNO = EM.EM_GRSNO      -- 來文電子交換檔
    LEFT JOIN mnda.dbo.DCS1_EMAL_FILE EF ON EM.EM_FID = EF.EF_ID                -- 來文電子資料檔
WHERE
--    TM.GRSNO = '1140020708'
--      TD.TD_FORMAT IN ('簽呈', '便籤')
      TM_RSTP in ('收')
--    AND TM.TM_SUBJ LIKE '%獎金%'
      AND TM.TM_NAME IN ('姚承佑')
--      AND TM.TM_PSID = 'M122255948'  -- 使用等於比IN更精確
--    AND EM.EM_GRSNO = '1150001261'
--    AND TD.TD_FORMAT = '檔案'
